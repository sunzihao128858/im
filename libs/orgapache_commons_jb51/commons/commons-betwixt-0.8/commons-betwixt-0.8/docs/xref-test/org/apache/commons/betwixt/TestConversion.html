31  public class TestConversion extends TestCase{
32  
33      public TestConversion(String name) {   
34          super(name);
35      }
36  
37      /***
38       * Betwixt does not (by default) try to convert nulls and empty strings
39       * @throws Exception
40       */
41      public void testNullTimestampConversion() throws Exception {
42          String xml = "<?xml version='1.0'?>" +
43              "<EventBean>" +
44              "<type>WARNING</type>" +
45              "<start>2004-02-10 00:00:00.0</start>" +
46              "<end/>" +
47              "</EventBean>";
48              
49         StringReader in = new StringReader(xml);
50         BeanReader reader = new BeanReader();
51         reader.registerBeanClass(EventBean.class);
52         EventBean bean = (EventBean) reader.parse(in);
53         
54         assertNotNull("Parsing should work", bean);
55         assertEquals("Type property", "WARNING", bean.getType());
56         assertEquals("Start property", "2004-02-10 00:00:00.0", bean.getStart().toString());
57         assertNull("End property", bean.getEnd());
58      }
